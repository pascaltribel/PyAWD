 20def solve_scalar_pde(grid: dvt.Grid, nx: int, ndt: int, ddt: float, epicenter: np.ndarray, velocity_model: dvt.Function) -> np.ndarray:
 21    """
 22    Solves the Acoustic Wave Equation for the input parameters
 23    Args:
 24        grid (dvt.Grid): A Devito Grid Object
 25        nx (int): The discretisation size of the array
 26        ndt (int): The number of iteration for which the result is stored
 27        ddt (float): The time step used for the Operator solving iterations
 28        epicenter (np.ndarray): The epicenter of the Ricker Wavelet at the beginning of the simulation
 29        velocity_model (devito.Function): The velocity field across which the wave propagates
 30    Returns:
 31        (numpy.ndarray): A numpy array containing the solutions for the `ndt` simulation steps
 32    """
 33    u = dvt.TimeFunction(name='u', grid=grid, space_order=2, save=ndt, time_order=2)
 34    u.data[:] = get_ricker_wavelet(nx, x0=epicenter[0], y0=epicenter[1])
 35    eq = dvt.Eq(u.dt2, (velocity_model ** 2) * (u.dx2 + u.dy2))
 36    stencil = dvt.solve(eq, u.forward)
 37    op = dvt.Operator(dvt.Eq(u.forward, stencil), opt='noop')
 38    op.apply(dt=ddt)
 39    return np.array(u.data)

 76    def __init__(self, size: int, nx: int = 128, sx: float = 1., ddt: float = 0.01, dt: int = 2, t: float = 10,
 77                 interrogators: List[Tuple] = None):
 78        """
 79        Args:
 80            size (int): The number of samples to generate in the dataset
 81            nx (int): The discretisation size of the array (maximum size is currently 955)
 82            sx (float): The sub-scaling factor of the array (0.5 means $\\frac{1}{2}$ values are returned)
 83            ddt (float): The time step used for the Operator solving iterations
 84            dt (float): The time step used for storing the wave propagation step (this should be higher than ddt)
 85            t (float): The simulations duration
 86            interrogators (List[Tuple]): A list containing the coordinates of each interrogator
 87        """
 88        if interrogators is None:
 89            interrogators = [(0, 0)]
 90        try:
 91            if dt < ddt:
 92                raise ValueError('dt should be >= ddt')
 93            self.size = size
 94            self.nx = min(nx, 955)
 95            self.sx = sx
 96            self.ddt = ddt
 97            self.dt = dt
 98            self.nt = int(t / self.dt)
 99            self.ndt = int(self.nt * (self.dt / self.ddt))
100            self.interrogators = interrogators
101            self.interrogators_data = None
102
103            self.grid = dvt.Grid(shape=(self.nx, self.nx), extent=(1., 1.))
104            self.velocity_model = dvt.Function(name='c', grid=self.grid)
105            self.velocity_model.data[:] = Marmousi(self.nx).get_data().T
106
107            self.epicenters = np.random.randint(-self.nx // 2, self.nx // 2, size=(self.size, 2)).reshape(
108                (self.size, 2))
109            self.data = None
110            self.generate_data()
111
112            self.cmap = get_black_cmap()
113
114        except ValueError as err:
115            print(err)

117    def generate_data(self):
118        """
119        Generates the dataset content by solving the Acoustic Wave PDE for each of the `epicenters`
120        """
121        self.data = []
122        self.interrogators_data = {interrogator: [] for interrogator in self.interrogators}
123        for i in tqdm(range(self.size)):
124            data = solve_scalar_pde(self.grid, self.nx, self.ndt, self.ddt, self.epicenters[i], self.velocity_model)
125            self.data.append(data[::int(self.ndt / self.nt)])
126            for interrogator in self.interrogators:
127                self.interrogators_data[interrogator].append(
128                    data[:, interrogator[0] + (self.nx // 2), -interrogator[1] + (self.nx // 2)])
129        self.data = np.array(self.data)
